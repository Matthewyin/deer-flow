# 视频 5：业务基线监控系统

> **时长**: 40 秒（1200 frames @30fps）
> **目标受众**: 业务运维、SRE、运营分析师

## Scene 间组装方式

主 Composition（`videos/BusinessBaseline/index.tsx`）使用 `<TransitionSeries>` 串联所有 Scene：

```tsx
import { TransitionSeries, linearTiming } from "@remotion/transitions";
import { fade } from "@remotion/transitions/fade";
import { slide } from "@remotion/transitions/slide";

const SCENES = [
  { Component: Intro,                frames: 90  },
  { Component: ProblemStatement,     frames: 150 },
  { Component: ThreeLayerArchitecture, frames: 210 },
  { Component: TwentySevenMetrics,   frames: 220 },
  { Component: DailyPipeline,        frames: 220 },
  { Component: BaselineComparisonChart, frames: 210 },
  { Component: SixMcpTools,          frames: 60  },
  { Component: Outro,                frames: 100 },
];

const TRANSITION = linearTiming({ durationInFrames: 10 });

export const BusinessBaselineVideo: React.FC = () => {
  return (
    <TransitionSeries>
      {SCENES.map((scene, i) => (
        <React.Fragment key={i}>
          <TransitionSeries.Sequence durationInFrames={scene.frames}>
            <scene.Component />
          </TransitionSeries.Sequence>
          {i < SCENES.length - 1 && (
            <TransitionSeries.Transition
              presentation={i % 2 === 0 ? fade() : slide({ direction: "from-right" })}
              timing={TRANSITION}
            />
          )}
        </React.Fragment>
      ))}
    </TransitionSeries>
  );
};
```

> 注意：7 个转场各 10 帧 = 70 帧重叠。总帧需精确控制为 1200 帧 = 40 秒 ✅
