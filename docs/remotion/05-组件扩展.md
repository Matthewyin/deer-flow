## TransitionSeries 组装代码

```tsx
import { TransitionSeries, linearTiming } from "@remotion/transitions";
import { fade } from "@remotion/transitions/fade";
import { slide } from "@remotion/transitions/slide";

// Scene 组件
import { Intro } from "./scenes/Intro";
import { ToolEcosystem } from "./scenes/ToolEcosystem";
import { DirectoryStructure } from "./scenes/DirectoryStructure";
import { ServerToolImpl } from "./scenes/ServerToolImpl";
import { RegisterConfig } from "./scenes/RegisterConfig";
import { DockerMount } from "./scenes/DockerMount";
import { EndToEndFlow } from "./scenes/EndToEndFlow";
import { Outro } from "./scenes/Outro";

// 各 scene 的帧数
const SCENES = [
  { Component: Intro,              frames: 90  },
  { Component: ToolEcosystem,      frames: 150 },
  { Component: DirectoryStructure, frames: 180 },
  { Component: ServerToolImpl,     frames: 180 },
  { Component: RegisterConfig,     frames: 150 },
  { Component: DockerMount,        frames: 150 },
  { Component: EndToEndFlow,       frames: 210 },
  { Component: Outro,              frames: 90  },
];

const TRANSITION = linearTiming({ durationInFrames: 10 });

export const ExtensionGuideVideo: React.FC = () => {
  return (
    <TransitionSeries>
      {SCENES.map((scene, i) => (
        <React.Fragment key={i}>
          <TransitionSeries.Sequence durationInFrames={scene.frames}>
            <scene.Component />
          </TransitionSeries.Sequence>
          {i < SCENES.length - 1 && (
            <TransitionSeries.Transition
              presentation={i % 2 === 0 ? fade() : slide({ direction: "from-right" })}
              timing={TRANSITION}
            />
          )}
        </React.Fragment>
      ))}
    </TransitionSeries>
  );
};
```

> 注意：7 个转场各 10 帧 = 70 帧重叠。总帧 = sum(scenes) - 70 = 1200 - 70 = 1130 帧（约 37.7 秒画面）。
> 但 `<TransitionSeries>` 的转场帧与相邻 Scene 重叠，所以 Composition 总时长仍然是 1200 帧 = 40 秒。
